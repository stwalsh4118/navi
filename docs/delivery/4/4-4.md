## Implementation Plan

Implement in `model.go`:
```go
func (m Model) renderHeader() string {
    title := "Claude Sessions"
    count := fmt.Sprintf("%d active", len(m.sessions))

    // Calculate padding for count on right
    padding := m.width - len(title) - len(count) - 4 // 4 for border padding
    if padding < 1 {
        padding = 1
    }

    content := title + strings.Repeat(" ", padding) + count
    return boxStyle.Width(m.width - 2).Render(content)
}

func (m Model) renderFooter() string {
    help := "↑/↓ navigate  ⏎ attach  d dismiss  q quit  r refresh"
    return boxStyle.Width(m.width - 2).Render(help)
}
```
